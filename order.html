function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = JSON.parse(e.postData.contents); let screenshotUrl = ''; // Upload screenshot to Drive if provided if (data.screenshot) { const folder = DriveApp.getFolderById('root'); const blob = Utilities.newBlob(Utilities.base64Decode(data.screenshot.split(',')[1]), 'image/jpeg', 'Order_' + data.trx + '_' + new Date().getTime() + '.jpg'); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); screenshotUrl = file.getUrl(); } // Add row to Sheet - 13 columns total sheet.appendRow([ new Date(), data.product, data.price, data.qty, data.color, data.name, data.phone, data.email, data.address, data.trx, screenshotUrl, 'Pending', // Status column L '' // Reminder_Sent column M ]); // Email to Customer: Order Received const customerSubject = 'Order Received - Haider Commercial Traders'; const customerBody = ` Hi ${data.name}, We have received your order details for: ${data.product} Order Details: Product: ${data.product} Quantity: ${data.qty} Amount: Rs ${data.price * data.qty} Transaction ID: ${data.trx} Screenshot: ${screenshotUrl? 'Uploaded' : 'Not provided'} Our team is now verifying your payment with Easypaisa. You will receive a confirmation email within 30 minutes once verified. If Transaction ID is incorrect, we will contact you on WhatsApp: ${data.phone} Thank you for shopping with Haider Commercial Traders! Need help? WhatsApp: 03471881297 `; MailApp.sendEmail(data.email, customerSubject, customerBody); // Email to Admin const adminSubject = 'New Order - Verify Payment - ' + data.name; const adminBody = ` New order received. Verify Transaction ID now. Product: ${data.product} Amount: Rs ${data.price * data.qty} Customer: ${data.name} Phone: ${data.phone} Email: ${data.email} TRX ID: ${data.trx} Screenshot: ${screenshotUrl || 'Not uploaded'} Address: ${data.address} Check Easypaisa app and update Status column in Sheet to Approved or Rejected. Sheet Link: ${SpreadsheetApp.getActiveSpreadsheet().getUrl()} `; MailApp.sendEmail('zebii3711@gmail.com', adminSubject, adminBody); return ContentService.createTextOutput(JSON.stringify({result: 'success'})).setMimeType(ContentService.MimeType.JSON); } // Trigger when Status is changed to Approved or Rejected function onEdit(e) { const sheet = e.source.getActiveSheet(); const range = e.range; // Status column is 12, Reminder_Sent is 13 if (range.getColumn() == 12 && range.getRow() > 1) { const row = range.getRow(); const status = range.getValue(); const email = sheet.getRange(row, 8).getValue(); const name = sheet.getRange(row, 6).getValue(); const product = sheet.getRange(row, 2).getValue(); const phone = sheet.getRange(row, 7).getValue(); if (status == 'Approved') { MailApp.sendEmail(email, 'Payment Successful - Order Confirmed ✅', ` Hi ${name}, Great news! Your payment has been verified successfully. Order: ${product} Status: CONFIRMED ✅ We are now preparing your order for dispatch. You will receive tracking details on WhatsApp within 2 hours. Thank you for shopping with Haider Commercial Traders! WhatsApp: 03471881297 `); } if (status == 'Rejected') { // First reminder after 5 minutes ScriptApp.newTrigger('sendRejectEmail') .timeBased() .after(5 * 60 * 1000) // 5 minutes .create(); // Store data for emails PropertiesService.getScriptProperties().setProperty('rejectEmail_' + row, email); PropertiesService.getScriptProperties().setProperty('rejectName_' + row, name); PropertiesService.getScriptProperties().setProperty('rejectProduct_' + row, product); PropertiesService.getScriptProperties().setProperty('rejectPhone_' + row, phone); PropertiesService.getScriptProperties().setProperty('rejectRow_' + row, row); } } } // First rejection email - 5 minutes after function sendRejectEmail() { const props = PropertiesService.getScriptProperties(); const allProps = props.getProperties(); for (let key in allProps) { if (key.startsWith('rejectEmail_')) { const row = key.split('_')[1]; const email = allProps['rejectEmail_' + row]; const name = allProps['rejectName_' + row]; const product = allProps['rejectProduct_' + row]; const phone = allProps['rejectPhone_' + row]; if (email) { MailApp.sendEmail(email, 'Payment Failed - Action Required ❌', ` Hi ${name}, We could not verify your payment for: ${product} Reason: Transaction ID not found or amount mismatch. Please verify your Transaction ID and contact us on WhatsApp: 03471881297 We will help you complete your order. If we do not hear from you, your order will be cancelled automatically. Haider Commercial Traders `); // Mark first reminder sent and schedule second reminder const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); sheet.getRange(row, 13).setValue('1st Reminder Sent'); // Second reminder after 30 minutes total ScriptApp.newTrigger('sendSecondRejectEmail') .timeBased() .after(25 * 60 * 1000) // 25 more minutes = 30 total .create(); props.deleteProperty('rejectEmail_' + row); // Keep other props for second email } } } // Delete this trigger const triggers = ScriptApp.getProjectTriggers(); for (let i = 0; i < triggers.length; i++) { if (triggers[i].getHandlerFunction() == 'sendRejectEmail') { ScriptApp.deleteTrigger(triggers[i]); } } } // Second rejection email - 30 minutes after rejection function sendSecondRejectEmail() { const props = PropertiesService.getScriptProperties(); const allProps = props.getProperties(); for (let key in allProps) { if (key.startsWith('rejectName_')) { const row = key.split('_')[1]; const email = allProps['rejectEmail_' + row]; const name = allProps['rejectName_' + row]; const product = allProps['rejectProduct_' + row]; const phone = allProps['rejectPhone_' + row]; if (email) { MailApp.sendEmail(email, 'Order Cancelled - Payment Not Verified', ` Hi ${name}, Your order for: ${product} has been cancelled. Reason: We did not receive valid payment verification within 30 minutes. If you believe this is an error, please contact us immediately on WhatsApp: 03471881297 with your payment proof. You can place a new order anytime at haider-traders.com Haider Commercial Traders `); // Mark final status const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); sheet.getRange(row, 13).setValue('Cancelled - No Response'); sheet.getRange(row, 12).setValue('Cancelled'); // Clean up properties props.deleteProperty('rejectEmail_' + row); props.deleteProperty('rejectName_' + row); props.deleteProperty('rejectProduct_' + row); props.deleteProperty('rejectPhone_' + row); props.deleteProperty('rejectRow_' + row); } } } // Delete this trigger const triggers = ScriptApp.getProjectTriggers(); for (let i = 0; i < triggers.length; i++) { if (triggers[i].getHandlerFunction() == 'sendSecondRejectEmail') { ScriptApp.deleteTrigger(triggers[i]); } } }
